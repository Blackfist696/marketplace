-- Création de la base de données
CREATE DATABASE IF NOT EXISTS marketplace_artisanal
  CHARACTER SET utf8mb4 
  COLLATE utf8mb4_unicode_ci;

USE marketplace_artisanal;

-- Table de référence : type_adresse
CREATE TABLE type_adresse (
  id_type_adresse INT AUTO_INCREMENT PRIMARY KEY,
  libelle VARCHAR(50) NOT NULL UNIQUE
);

INSERT INTO type_adresse (libelle) VALUES ('livraison'), ('facturation');

-- Table de référence : statut_commande
CREATE TABLE statut_commande (
  id_statut_commande INT AUTO_INCREMENT PRIMARY KEY,
  libelle VARCHAR(50) NOT NULL UNIQUE
);

INSERT INTO statut_commande (libelle) VALUES
  ('en_attente'), ('payee'), ('en_preparation'), ('expediee'), ('livree'), ('annulee');

-- Table de référence : methode_paiement
CREATE TABLE methode_paiement (
  id_methode_paiement INT AUTO_INCREMENT PRIMARY KEY,
  libelle VARCHAR(50) NOT NULL UNIQUE
);

INSERT INTO methode_paiement (libelle) VALUES ('carte'), ('paypal'), ('virement');
-- Table : role
CREATE TABLE role (
  id_role INT AUTO_INCREMENT PRIMARY KEY,
  nom VARCHAR(50) NOT NULL UNIQUE,
  description TEXT
);

INSERT INTO role (nom, description) VALUES
  ('anonyme',        'Accès public au catalogue sans connexion'),
  ('client',         'Achat de produits et suivi de commandes'),
  ('artisan',        'Gestion autonome de sa boutique en ligne'),
  ('administrateur', 'Gestion complète de la plateforme');

-- Table : utilisateur
CREATE TABLE utilisateur (
  id_utilisateur INT AUTO_INCREMENT PRIMARY KEY,
  email VARCHAR(150) NOT NULL UNIQUE,
  mot_de_passe VARCHAR(255) NOT NULL,
  id_role INT NOT NULL,
  date_inscription DATETIME DEFAULT CURRENT_TIMESTAMP,
  actif BOOLEAN DEFAULT TRUE,
  FOREIGN KEY (id_role) REFERENCES role(id_role)
);

-- Table : personne
CREATE TABLE personne (
  id_personne INT AUTO_INCREMENT PRIMARY KEY,
  id_utilisateur INT NOT NULL UNIQUE,
  nom VARCHAR(100) NOT NULL,
  prenom VARCHAR(100) NOT NULL,
  telephone VARCHAR(20),
  FOREIGN KEY (id_utilisateur) REFERENCES utilisateur(id_utilisateur)
);

-- Table : adresse
CREATE TABLE adresse (
  id_adresse INT AUTO_INCREMENT PRIMARY KEY,
  id_personne INT NOT NULL,
  id_type_adresse INT NOT NULL,
  rue VARCHAR(255) NOT NULL,
  numero VARCHAR(10) NOT NULL,
  code_postal VARCHAR(10) NOT NULL,
  ville VARCHAR(100) NOT NULL,
  pays VARCHAR(100) NOT NULL DEFAULT 'Belgique',
  principale BOOLEAN DEFAULT FALSE,
  FOREIGN KEY (id_personne) REFERENCES personne(id_personne),
  FOREIGN KEY (id_type_adresse) REFERENCES type_adresse(id_type_adresse)
);

-- Table : template_css
CREATE TABLE template_css (
  id_template INT AUTO_INCREMENT PRIMARY KEY,
  nom VARCHAR(100) NOT NULL,
  description TEXT,
  fichier_css VARCHAR(255) NOT NULL,
  apercu VARCHAR(255)
);

INSERT INTO template_css (nom, description, fichier_css, apercu) VALUES
  ('Nature',    'Tons chauds dorés, inspiration apicole et naturelle',  'nature.css',    'apercu_nature.png'),
  ('Elegance',  'Lignes épurées, tons crème et noir pour un style sobre', 'elegance.css',  'apercu_elegance.png'),
  ('Moderne',   'Design contemporain avec couleurs vives et dynamiques', 'moderne.css',   'apercu_moderne.png'),
  ('Rustique',  'Aspect artisanal traditionnel et chaleureux',           'rustique.css',  'apercu_rustique.png');

-- Table : artisan
CREATE TABLE artisan (
  id_artisan INT AUTO_INCREMENT PRIMARY KEY,
  id_utilisateur INT NOT NULL UNIQUE,
  id_template INT,
  nom_boutique VARCHAR(150) NOT NULL,
  description TEXT,
  logo VARCHAR(255),
  banniere VARCHAR(255),
  tva VARCHAR(14) UNIQUE,
  iban VARCHAR(34),
  couleur_primaire VARCHAR(7) DEFAULT '#D4A855',
  couleur_secondaire VARCHAR(7),
  police_principale VARCHAR(100),
  commission DECIMAL(5,2) DEFAULT 5.00,
  valide BOOLEAN DEFAULT FALSE,
  date_validation DATETIME,
  FOREIGN KEY (id_utilisateur) REFERENCES utilisateur(id_utilisateur),
  FOREIGN KEY (id_template) REFERENCES template_css(id_template)
);

-- Table : categorie
CREATE TABLE categorie (
  id_categorie INT AUTO_INCREMENT PRIMARY KEY,
  id_categorie_parent INT NULL,
  nom VARCHAR(100) NOT NULL,
  description TEXT,
  image VARCHAR(255),
  ordre INT DEFAULT 0,
  actif BOOLEAN DEFAULT TRUE,
  FOREIGN KEY (id_categorie_parent) REFERENCES categorie(id_categorie)
);

INSERT INTO categorie (nom, description, ordre, actif) VALUES
  ('Miels',        'Miels monofloral, toutes fleurs et miels de terroir',           1, TRUE),
  ('Savons',       'Savons artisanaux saponifiés à froid, 100% naturels',           2, TRUE),
  ('Confiseries',  'Bonbons, nougats, caramels et douceurs à base de miel',        3, TRUE),
  ('Cosmétiques',  'Crèmes, baumes, soins du visage et du corps aux ingrédients naturels', 4, TRUE),
  ('Bougies',      'Bougies en cire d\'abeille pure, fabriquées artisanalement',    5, TRUE);

-- Table : produit
CREATE TABLE produit (
  id_produit INT AUTO_INCREMENT PRIMARY KEY,
  id_artisan INT NOT NULL,
  id_categorie INT NOT NULL,
  nom VARCHAR(200) NOT NULL,
  description TEXT,
  prix_ht DECIMAL(10,2) NOT NULL,
  taux_tva DECIMAL(5,2) DEFAULT 6.00,
  stock INT DEFAULT 0,
  poids DECIMAL(8,2),
  image_principale VARCHAR(255),
  actif BOOLEAN DEFAULT TRUE,
  mis_en_avant BOOLEAN DEFAULT FALSE,
  nb_vues INT DEFAULT 0,
  date_creation DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (id_artisan) REFERENCES artisan(id_artisan),
  FOREIGN KEY (id_categorie) REFERENCES categorie(id_categorie)
);

-- Table : image_produit
CREATE TABLE image_produit (
  id_image INT AUTO_INCREMENT PRIMARY KEY,
  id_produit INT NOT NULL,
  chemin VARCHAR(255) NOT NULL,
  alt VARCHAR(200),
  ordre INT DEFAULT 0,
  FOREIGN KEY (id_produit) REFERENCES produit(id_produit)
);

-- Table : panier
CREATE TABLE panier (
  id_panier INT AUTO_INCREMENT PRIMARY KEY,
  id_utilisateur INT NULL,
  session_id VARCHAR(255),
  date_creation DATETIME DEFAULT CURRENT_TIMESTAMP,
  date_modification DATETIME ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (id_utilisateur) REFERENCES utilisateur(id_utilisateur)
);

-- Table : ligne_panier
CREATE TABLE ligne_panier (
  id_ligne INT AUTO_INCREMENT PRIMARY KEY,
  id_panier INT NOT NULL,
  id_produit INT NOT NULL,
  quantite INT NOT NULL DEFAULT 1,
  prix_unitaire DECIMAL(10,2) NOT NULL,
  FOREIGN KEY (id_panier) REFERENCES panier(id_panier),
  FOREIGN KEY (id_produit) REFERENCES produit(id_produit)
);

-- Table : commande
CREATE TABLE commande (
  id_commande INT AUTO_INCREMENT PRIMARY KEY,
  reference VARCHAR(50) NOT NULL UNIQUE,
  id_utilisateur INT NOT NULL,
  id_adresse_livraison INT NOT NULL,
  id_adresse_facturation INT NOT NULL,
  id_statut_commande INT NOT NULL,
  total_ht DECIMAL(10,2) NOT NULL,
  total_tva DECIMAL(10,2) NOT NULL,
  frais_livraison DECIMAL(10,2) DEFAULT 0.00,
  total_ttc DECIMAL(10,2) NOT NULL,
  date_commande DATETIME DEFAULT CURRENT_TIMESTAMP,
  date_paiement DATETIME,
  FOREIGN KEY (id_utilisateur) REFERENCES utilisateur(id_utilisateur),
  FOREIGN KEY (id_adresse_livraison) REFERENCES adresse(id_adresse),
  FOREIGN KEY (id_adresse_facturation) REFERENCES adresse(id_adresse),
  FOREIGN KEY (id_statut_commande) REFERENCES statut_commande(id_statut_commande)
);

-- Table : ligne_commande
CREATE TABLE ligne_commande (
  id_ligne INT AUTO_INCREMENT PRIMARY KEY,
  id_commande INT NOT NULL,
  id_produit INT NOT NULL,
  id_artisan INT NOT NULL,
  quantite INT NOT NULL,
  prix_unitaire_ht DECIMAL(10,2) NOT NULL,
  taux_tva DECIMAL(5,2) NOT NULL,
  FOREIGN KEY (id_commande) REFERENCES commande(id_commande),
  FOREIGN KEY (id_produit) REFERENCES produit(id_produit),
  FOREIGN KEY (id_artisan) REFERENCES artisan(id_artisan)
);

-- Table : paiement
CREATE TABLE paiement (
  id_paiement INT AUTO_INCREMENT PRIMARY KEY,
  id_commande INT NOT NULL UNIQUE,
  id_methode_paiement INT NOT NULL,
  reference_externe VARCHAR(255),
  montant DECIMAL(10,2) NOT NULL,
  statut VARCHAR(50) NOT NULL,
  date_paiement DATETIME,
  FOREIGN KEY (id_commande) REFERENCES commande(id_commande),
  FOREIGN KEY (id_methode_paiement) REFERENCES methode_paiement(id_methode_paiement)
);

-- Table : avis
CREATE TABLE avis (
  id_avis INT AUTO_INCREMENT PRIMARY KEY,
  id_produit INT NOT NULL,
  id_utilisateur INT NOT NULL,
  note TINYINT NOT NULL CHECK (note BETWEEN 1 AND 5),
  commentaire TEXT,
  date_avis DATETIME DEFAULT CURRENT_TIMESTAMP,
  valide BOOLEAN DEFAULT FALSE,
  FOREIGN KEY (id_produit) REFERENCES produit(id_produit),
  FOREIGN KEY (id_utilisateur) REFERENCES utilisateur(id_utilisateur)
);

-- Table : statistique_artisan
CREATE TABLE statistique_artisan (
  id_stat INT AUTO_INCREMENT PRIMARY KEY,
  id_artisan INT NOT NULL,
  date DATE NOT NULL,
  nb_visites INT DEFAULT 0,
  nb_commandes INT DEFAULT 0,
  ca_ht DECIMAL(10,2) DEFAULT 0.00,
  panier_moyen DECIMAL(10,2) DEFAULT 0.00,
  UNIQUE KEY uk_stat_artisan_date (id_artisan, date),
  FOREIGN KEY (id_artisan) REFERENCES artisan(id_artisan)
);
